{{/* ──────────────────────────────────────────────────────────────────── SWIMLANE MINI — single-row chart driven by pre-computed event records. Designed for actor term pages. Reads from data/actor_events.json instead of scanning site.RegularPages, so per-page render is O(events-for-this-actor) not O(all-events). Records come in pre-shaped: [{slug, date, lane, imp, stat, title}, ...] with `lane` already mapped to one of the 12 consolidated lane keys. Inputs (passed via dict): label: string the row title (e.g. actor name) records: []record pre-computed event records fromYear: int x-axis start toYear: int x-axis end chart_id: string unique DOM id ──────────────────────────────────────────────────────────────────── */}} {{- $label := .label | default "" -}} {{- $records := .records | default slice -}} {{- $fromYear := .fromYear | default 2000 -}} {{- $toYear := .toYear | default 2026 -}} {{- $chartId := .chart_id | default "swim-mini" -}} {{- if $records -}} {{- $W := 1200 -}} {{- $rowH := 56 -}} {{- $padTop := 30 -}} {{- $padBottom := 14 -}} {{- $padLeft := 220 -}} {{- $padRight := 18 -}} {{- $H := add (add $padTop $padBottom) $rowH -}} {{- $rowY := add $padTop (mul 0.5 $rowH) -}} {{- $plotW := sub $W (add $padLeft $padRight) -}} {{- $yearSpan := add 1 (sub $toYear $fromYear) -}} {{- $colW := div (float $plotW) (float $yearSpan) -}} {{- $laneColors := dict "regulatory" "#7a1f1f" "financial" "#a3406b" "intelligence" "#3a5a7a" "corruption" "#b08833" "executive" "#5c2a4a" "legislative" "#5a7a3a" "judicial" "#9a5a1a" "media" "#2a6a6a" "democratic" "#b56a4a" "military" "#3a3a3a" "labor" "#7a7a3a" "other" "#8a8273" -}} {{- /* Density-adaptive radius scaling */ -}} {{- $count := len $records -}} {{- $densityTarget := 80 -}} {{- $densityFactor := 1.0 -}} {{- if gt $count $densityTarget -}} {{- $densityFactor = math.Sqrt (div (float $densityTarget) (float $count)) -}} {{- if lt $densityFactor 0.35 -}}{{- $densityFactor = 0.35 -}}{{- end -}} {{- end -}} {{- end -}}
